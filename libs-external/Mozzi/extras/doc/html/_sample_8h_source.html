182  char next() { // 4us
183 
184  if (phase_fractional>endpos_fractional){
185  if (looping) {
186  phase_fractional = startpos_fractional + (phase_fractional - endpos_fractional);
187  }else{
188  return 0;
189  }
190  }
191  char out;
192  if(INTERP==INTERP_LINEAR){
193  // WARNNG this is hard coded for when SAMPLE_F_BITS is 16
194  unsigned int index = phase_fractional >> SAMPLE_F_BITS;
195  out = (char)pgm_read_byte_near(table + index);
196  char difference = (char)pgm_read_byte_near((table + 1) + index) - out;
197  char diff_fraction = (char)(((((unsigned int) phase_fractional)>>8)*difference)>>8); // (unsigned int) phase_fractional keeps low word, then>> for only 8 bit precision
198  out += diff_fraction;
199  }else{
200  out = (char)pgm_read_byte_near(table + (phase_fractional >> SAMPLE_F_BITS));
201  }
202  incrementPhase();
203  return out;
204  }
