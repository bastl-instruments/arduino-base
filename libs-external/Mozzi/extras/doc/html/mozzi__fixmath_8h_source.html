369  volatile unsigned char zeroReg=0;
370  volatile unsigned int multipliedCounter=oscillators[c].phaseStep;
371  asm volatile
372  (
373  // high unsigned chars mult together = high unsigned char
374  "ldi %A[outVal],0" "\n\t"
375  "mul %B[phaseStep],%B[pitchB}]" "\n\t"
376  "mov %B[outVal],r0" "\n\t"
377  // ignore overflow into r1 (should never overflow)

379  "mul %A[phaseStep],%B[pitchB}]" "\n\t"

383  // high unsigned char* low unsigned char -> both unsigned chars
384  "mul %B[phaseStep],%A[pitchB}]" "\n\t"
385  "add %A[outVal],r0" "\n\t"
386  // carry into high unsigned char
387  "adc %B[outVal],r1" "\n\t"
388  // low unsigned char * low unsigned char -> round
