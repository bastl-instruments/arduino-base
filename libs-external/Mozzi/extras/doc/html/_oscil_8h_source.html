173  // PM: cos((angle += incr) + change)
174  // FM: cos(angle += (incr + change))
175  // The ratio of deviation to modulation frequency is called the "index of modulation". ( I = d / Fm )
176  inline
177  char phMod(Q15n16 phmod_proportion)
178  {
179  incrementPhase();
180  return (char)pgm_read_byte_near(table + (((phase_fractional+(phmod_proportion * NUM_TABLE_CELLS))>>OSCIL_F_BITS) & (NUM_TABLE_CELLS - 1)));
181  }
