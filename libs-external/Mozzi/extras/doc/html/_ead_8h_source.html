124  inline
125  unsigned char next()
126  {
127  if(attack_phase)
128  {
129  // signed multiply A(a1,b1) * A(a2,b2)=A(a1 +a2 +1,b1 +b2)
130  Q8n24state += ((Q8n24_FIX1 - Q8n24state) * Q8n8attack) >> 8; // Q8n24, shifts all back into n24
131  if (Q8n24state >= Q8n24_FIX1-256)
132  {
133  Q8n24state = Q8n24_FIX1-256;
134  attack_phase = false;
135  }
136  }else{ /* decay phase */
137  Q8n24state -= (Q8n24state * Q8n8decay)>>8;
138  }
139  return Q8n24_to_Q0n8(Q8n24state);
140  }

152  /* convert milliseconds to 1-p, with p a real pole */
153  inline
154  float millisToOneMinusRealPole(unsigned int milliseconds)
155  {
156  static const float NUMERATOR = 1000.0f * log(0.001f);
157  return -expm1(NUMERATOR / ((float)UPDATE_RATE * milliseconds));
158  }

161  // Compute exp(x) - 1 without loss of precision for small values of x.
162  inline
163  float expm1(float x)
164  {
165  if (fabs(x) < 1e-5)
166  {
167  return x + 0.5*x*x;
168  }
169  else
170  {
171  return exp(x) - 1.0;
172  }
