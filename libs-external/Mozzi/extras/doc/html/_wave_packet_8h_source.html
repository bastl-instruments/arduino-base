124  int next()
125  {
126  gcentrefreq = aCentrefreq.next();
127  gbandwidth = aBandwidth.next();
128  int phase1 = aPhasor.next()>>16; // -0.5 to 0.5, 0n15

132  return signalPath(params1, phase1);

165  inline
166  int signalPath(struct parameters &param, int phase) // 25us? * 2
167  {
168  //setPin13High();
169  int index;
170 
171  if(phase<param.previous_phase)
172  {
173  param.centrefreq = gcentrefreq>>8;
174  param.bandwidth = Q15n16_to_Q23n8(gbandwidth);
175  }
176  param.previous_phase = phase;
177 
178  // oscillation
179  index = (param.centrefreq * phase)>>16;
180  // hack to make peak in middle of packet, otherwise it centres around a zero-crossing..
181  index += COS8192_NUM_CELLS>>1;
182  index &= COS8192_NUM_CELLS-1;
183  char sig1 = aCos.atIndex(index);
184 
185  // packet envelope
186  Q23n8 bwphase = (param.bandwidth * phase)>>8;
187  bwphase += COS8192_NUM_CELLS>>1;
188  index = constrain(bwphase, 0, (COS8192_NUM_CELLS-1));
189  unsigned char packet_width = 128 + aCos.atIndex(index);
190  // if (AUDIO_MODE == HIFI){
191  // return ((int) sig1 * packet_width)>>3; // scaled to fit HIFI updateAudio output
192  // }else{
193  // return ((int) sig1 * packet_width)>>8; // scaled to fit STANDARD updateAudio output
194  // }
195 
196  return ((int) sig1 * packet_width);
197  }
