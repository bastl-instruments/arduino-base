55 template <unsigned int CONTROL_UPDATE_RATE, unsigned int LERP_RATE>
56 class ADSR
57 {
58 private:
59 
60  const unsigned int LERPS_PER_CONTROL;
61 
62  unsigned int phase_control_step_counter;
63  unsigned int phase_num_control_steps;
64 
65  enum {ATTACK,DECAY,SUSTAIN,RELEASE,IDLE};
66 
67 
68  struct phase{
69  byte phase_type;
70  unsigned int control_steps;
71  unsigned long lerp_steps;
72  Q8n0 level;
73  }attack,decay,sustain,release,idle;
74 
75  phase * current_phase;
76 
77  // Linear audio rate transitions for envelope
78  Line <unsigned long> transition;
79 
80  inline
81  unsigned int convertMsecToControlSteps(unsigned int msec){
82  return (uint) (((ulong)msec*CONTROL_UPDATE_RATE)>>10); // approximate /1000 with shift
83  }
84 
85 
86  inline
87  void setPhase(phase * next_phase) {
88  phase_control_step_counter = 0;
89  phase_num_control_steps = next_phase->control_steps;
90  transition.set(Q8n0_to_Q16n16(next_phase->level),next_phase->lerp_steps);
91  current_phase = next_phase;
92  }
93 
94 
95 
96  inline
97  void checkForAndSetNextPhase(phase * next_phase) {
98  if (++phase_control_step_counter >= phase_num_control_steps){
99  setPhase(next_phase);
100  }
101  }
102 
103 
104  inline
105  void checkForAndSetIdle() {
106  if (++phase_control_step_counter >= phase_num_control_steps){
107  transition.set(0,0,1);
108  current_phase = &idle;
109  }
110  }
111 
112 
113 
114 inline
115  void setTime(phase * p, unsigned int msec)
116  {
117  p->control_steps=convertMsecToControlSteps(msec);
118  p->lerp_steps = (ulong) p->control_steps * LERPS_PER_CONTROL;
119  }
120 
121 
122 public:
123 
126  ADSR():LERPS_PER_CONTROL(LERP_RATE/CONTROL_UPDATE_RATE)
127  {
128  attack.phase_type = ATTACK;
129  decay.phase_type = DECAY;
130  sustain.phase_type = SUSTAIN;
131  release.phase_type = RELEASE;
132  idle.phase_type = IDLE;
133  release.level = 0;
134  }
135 
136 
137 
141  void update(){ // control rate
142 
143  switch(current_phase->phase_type) {
144 
145  case ATTACK:
146  checkForAndSetNextPhase(&decay);
147  break;
148 
149  case DECAY:
150  checkForAndSetNextPhase(&sustain);
151  break;
152 
153  case SUSTAIN:
154  checkForAndSetNextPhase(&release);
155  break;
156 
157  case RELEASE:
158  checkForAndSetIdle();
159  break;
160 
161  }
162  }
